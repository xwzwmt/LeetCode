```
    public void moveZeroes(int[] nums) {
        if(nums.length==0||nums==null){
            return;
        }
        int curPoint=0;
        for(int i=0;i<nums.length;i++){
            if(nums[i]!=0){
                nums[curPoint]=nums[i];
                curPoint++;
            }
        }
        while(curPoint<nums.length){
            nums[curPoint++]=0;
        }
    }
```
