保存每次递归的结果，避免重复计算

```
 public int climbStairs(int n) {
        if(n==1){
            return 1;
        }
        if(n==2){
            return 2;
        }
        int a[]=new int[n+1];
        a[1]=1;
        a[2]=2;
        return climb(a,n);
    }
     public int climb(int a[],int n){
        if(a[n]>0){
            return a[n];
        }
        a[n]=climb(a,n-1)+climb(a,n-2);
        return a[n];
    }
```
