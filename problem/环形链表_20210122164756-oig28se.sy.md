 public boolean hasCycle(ListNode head) {
        if(head==null){
            return  false;
        }
        HashSet<ListNode> set=new HashSet();
        ListNode node=head;
        while(node.next!=null){
            if(!set.add(node)){
                return true;
            }
            node=node.next;
        }
        return false;
    }
