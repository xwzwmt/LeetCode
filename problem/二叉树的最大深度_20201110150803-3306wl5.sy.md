```
public int maxDepth(TreeNode root) {
         if(root==null){
            return 0;
        }
        int le = 0,ri=0;
        if(root.left!=null){
           le=maxDepth(root.left);
        }
        if(root.right!=null){
            ri=maxDepth(root.right);
        }
       return Math.max(le,ri)+1;
    }
```

```
 int count=0;
        if(root==null){
            return count;
        }
        Queue<TreeNode> queue=new LinkedList<>();
        queue.offer(root);
        while (!queue.isEmpty()){
            int size=queue.size();
            while(size>0){
                TreeNode node=queue.poll();
                if(node.left!=null){
                    queue.offer(node.left);
                }
                if(node.right!=null){
                    queue.offer(node.right);
                }
                size--;
            }
            count++;
        }
        return count;
```
