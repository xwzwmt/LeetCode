public int[] twoSum(int[] nums, int target) {
 HashMap<Integer,Integer> map=new HashMap<>();
        int [] res=new int[2];
        for(int i=0;i<nums.length;i++){
            int value=map.getOrDefault(target-nums[i],-1);
            if(value!=-1){
                res[0]=i;
                res[1]=value;
                return res;
            }
            map.put(nums[i],i);
        }
        return res;
    }
